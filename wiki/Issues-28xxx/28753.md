def ram_type(self):
    ram_types=list()
    for b in self.branch_points():
        ram_inds=list()
        for c in F.preimage(b)
            ram_inds.append(e_self(c)) if e_self(c) >1
        ram_types.append(ram_inds)

    return ram_types
