```C++
#include <iostream>
#include <givaro/modular-floating.h>

using namespace std;

typedef Givaro::Modular<double> Modular_double;

int main(int argc, char** argv) {
  double zero, one;
  //double zero = 0, one = 0;
  long modulus = 2;
  Modular_double F(modulus);
  F.init(zero, (int)0);
  F.init(one, (int)1);

  cout << "zero: " << zero << "\n" << "one: " << one << "\n";
  return 0;
}
```
